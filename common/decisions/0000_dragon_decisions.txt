    desc = attempt_to_train_dragon_decision_desc

	major = yes

    # AI rule check
    ai_potential = {
        NOT = {
            has_game_rule = non_existent_ai_role
        }
    }
	ai_goal = yes
	ai_will_do = 100

    is_shown = {
        is_landed = yes
        is_adult = yes
		OR = {
			has_trait = novice_dragon_rider
			has_trait = dragon_rider
			has_trait = renowned_dragon_rider
		}
    }

    cost = {
        gold = 250
    }

    is_valid = {
        gold > 250

			has_trait = renowned_dragon_rider
		}
        NOT = {
            any_character_artifact = {
                artifact_type = dragon
            }
        }
    }

    effect = {
        root = { trigger_event = dragon_world_new_game.0003 }
        add_stress = 40
    }
}

recruit_dragon_rider_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_invite_knights.dds"
	sort_order = 75

	cooldown = { years = 3 }

    desc = recruit_dragon_rider_decision_decision_desc

    # AI rule check
    ai_potential = {
        NOT = {
            has_game_rule = non_existent_ai_role
        }
        is_landed = yes
        is_adult = yes
        OR = {
            has_trait = novice_dragon_rider
            has_trait = dragon_rider
            has_trait = renowned_dragon_rider
        }
    }
    ai_goal = yes

    ai_will_do = {
        base = 100

        modifier = {
            factor = 0
            any_courtier = {
                OR = {
                    has_trait = novice_dragon_rider
                    has_trait = dragon_rider
                    has_trait = renowned_dragon_rider
                }
            }
        }
    }

    is_shown = {
        is_landed = yes
        is_adult = yes
        OR = {
            has_trait = novice_dragon_rider
            has_trait = dragon_rider
            has_trait = renowned_dragon_rider
        }
    }

	cost = {
		gold = {
			value = minor_gold_value
		}
		prestige = {
			value = medium_prestige_value
		}
	}

    is_valid = {
        gold > 50
        AND = {
            OR = {
                has_trait = novice_dragon_rider
                has_trait = dragon_rider
                has_trait = renowned_dragon_rider
            }
        }
    }

    effect = {
        custom_tooltip = recruit_dragon_rider_specialist_decision.tt
        hidden_effect = {
            recruit_terrain_specialist_effect = {
                SPECIALIST_TRAIT = renowned_dragon_rider
                SPECIALIST_TEMPLATE_CHARACTER = dragon_rider_commander_character
            }
        }
    }
}
